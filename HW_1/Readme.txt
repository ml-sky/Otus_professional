#CentOs 7 с обновлением ядра до последней версии

#Запускаем Vagrnat file
#Если находимся в России то предварительно скачиваем с помощью браузерного VPN образ centos/7
#Добавление образа
vagrant box add --name 'centos/7'
#Проверяем что образ доступен 
vagrant box list | grep centos

#Если используем полноценный VPN то сразу запускаем следующую команду
vagrant up

#Vagrant поднимет ВМ и с помощью скрипта kernel-update.sh проверит стабильную версию ядра и обновит ядро, в grub перезапишет конфиг.
